--Consulta de las tiendas que han vendido más de 100 UND del producto 100 en los últimos 60 días.
SELECT B.SEDE_ID, B.NOMBRE_SEDE, B.TOTAL_UNIDADES_VENDIDAS
FROM
(SELECT A.SEDE_ID, (SELECT S.NOMBRE FROM HYSOTELO.SEDE S WHERE S.ID = A.SEDE_ID)NOMBRE_SEDE, SUM(A.UNIDADES)TOTAL_UNIDADES_VENDIDAS
FROM
(SELECT DISTINCT FA.ID, DM.UNIDADES, M.SEDE_ID SEDE_ID
FROM HYSOTELO.FACTURA FA, HYSOTELO.MANTENIMIENTO M, HYSOTELO.DETALLE_MANTENIMIENTO DM
WHERE FA.ESTADO = 'A' AND 
FA.MANTENIMIENTO_ID = M.ID AND M.ID = DM.MANTENIMIENTO_ID AND DM.ESTADO = 'A' AND DM.PRODUCTO_ID = 100 AND
TO_DATE(FA.FECHA_EMISION,'DD/MM/YYYY') >=  (SELECT TO_DATE(SYSDATE-60,'DD/MM/YYYY') FROM DUAL))A
GROUP BY SEDE_ID)B
WHERE B.TOTAL_UNIDADES_VENDIDAS > 100
